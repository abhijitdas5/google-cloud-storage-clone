async function login() { const email = document.getElementById("email").value const password = document.getElementById("password").value const res = await fetch("https://google-cloud-storage-clone.onrender.com/auth/login", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ email, password }) }) const data = await res.json() // 🔥 IMPORTANT localStorage.setItem("token", data.token) // redirect window.location.href = "/dashboard.html" }
